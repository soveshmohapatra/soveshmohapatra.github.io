Deconstructing Optimizers

Part 2: Fifteen Lines Each

By Sovesh Mohapatra • May 2026
Overview

Every optimizer is a class with two methods: .step() and .zero_grad(). No torch.optim inheritance, no abstract base class.

SGD

class SGD:
    def __init__(self, params, lr=1e-2):
        self.params = list(params); self.lr = lr
    def step(self):
        with torch.no_grad():
            for p in self.params:
                if p.grad is None: continue
                p.add_(p.grad, alpha=-self.lr)

Momentum

class Momentum:
    def __init__(self, params, lr=1e-2, momentum=0.9):
        self.params = list(params); self.lr = lr; self.momentum = momentum
        self.v = [torch.zeros_like(p) for p in self.params]
    def step(self):
        with torch.no_grad():
            for p, v in zip(self.params, self.v):
                v.mul_(self.momentum).add_(p.grad)
                p.add_(v, alpha=-self.lr)

Adam

class Adam:
    def __init__(self, params, lr=1e-3, betas=(0.9, 0.999), eps=1e-8):
        self.params = list(params); self.lr = lr
        self.beta1, self.beta2 = betas; self.eps = eps; self.t = 0
        self.m = [torch.zeros_like(p) for p in self.params]
        self.v = [torch.zeros_like(p) for p in self.params]
    def step(self):
        self.t += 1
        bc1 = 1 - self.beta1 ** self.t
        bc2 = 1 - self.beta2 ** self.t
        for p, m, v in zip(self.params, self.m, self.v):
            g = p.grad
            m.mul_(self.beta1).add_(g, alpha=1 - self.beta1)
            v.mul_(self.beta2).addcmul_(g, g, value=1 - self.beta2)
            p.addcdiv_(m/bc1, (v/bc2).sqrt().add_(self.eps), value=-self.lr)

AdamW

Identical to Adam except the parameter step:

p.mul_(1 - self.lr * self.wd)            # decoupled weight decay
p.addcdiv_(m/bc1, (v/bc2).sqrt().add_(self.eps), value=-self.lr)

Lion

class Lion:
    def __init__(self, params, lr=1e-4, betas=(0.9, 0.99), weight_decay=0.0):
        self.params = list(params); self.lr = lr
        self.beta1, self.beta2 = betas; self.wd = weight_decay
        self.m = [torch.zeros_like(p) for p in self.params]
    def step(self):
        for p, m in zip(self.params, self.m):
            g = p.grad
            c = self.beta1 * m + (1 - self.beta1) * g
            p.add_(torch.sign(c) + self.wd * p, alpha=-self.lr)
            m.mul_(self.beta2).add_(g, alpha=1 - self.beta2)

One buffer instead of two ($v$ is gone). Sign-based update direction. Momentum updated after the parameter step with a different $\beta$.

Summary

  • Five optimizers in ~75 lines of Python.
  • Each is a one-line modification of its predecessor.
  • No torch.optim inheritance required — the interface is two methods.
